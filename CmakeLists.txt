# version requirement of cmake
cmake_minimum_required(VERSION 3.10)

# project name and version
project(
    DS_demo         # project name
    VERSION 1.0     # project version
    LANGUAGES CXX   # C++ project declaration
)

# setting of C++ standard
set(CXX_STANDARD 17)
set(CXX_STANDARD_REQUIRED True)

# set .exe file output to root of project
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${PROJECT_SOURCE_DIR})

# define .exe file
add_executable(
    DS_demo
    src/main.cpp
)
